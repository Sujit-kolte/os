// 1. Show all databases
show dbs

// 2. Create or switch to a database
use myDatabase

// 3. Show current database
db

// 4. Create a collection and insert a document
db.users.insertOne({ name: "Alice", age: 25, city: "New York" })

// 5. Insert multiple documents
db.users.insertMany([
  { name: "Bob", age: 30, city: "Los Angeles" },
  { name: "Charlie", age: 28, city: "Chicago" }
])

// 6. Find all documents in a collection
db.users.find()

// 7. Find documents with a condition
db.users.find({ age: { $gt: 25 } })   // age greater than 25

// 8. Find documents with multiple conditions
db.users.find({ age: { $gt: 25 }, city: "Chicago" })

// 9. Find documents with regex
db.users.find({ name: { $regex: /^A/ } })  // name starts with "A"

// 10. Update a single document
db.users.updateOne({ name: "Alice" }, { $set: { age: 26 } })

// 11. Update multiple documents
db.users.updateMany({ city: "Chicago" }, { $set: { city: "San Francisco" } })

// 12. Delete a single document
db.users.deleteOne({ name: "Bob" })

// 13. Delete multiple documents
db.users.deleteMany({ age: { $lt: 30 } })

// 14. Count documents in a collection
db.users.countDocuments()

// 15. Drop a collection
db.users.drop()

// 16. Drop the current database
db.dropDatabase()
