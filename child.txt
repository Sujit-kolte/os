#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <stdlib.h>

void bubble_acs(int a[50], int n)
{
    int i, j, temp;
    for (i = n - 1; i > 0; i--)
    {
        for (j = 0; j < i; j++)
        {
            if (a[j] > a[j + 1])
            {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }
}

void bubble_dcs(int a[50], int n)
{
    int i, j, temp;
    for (i = n - 1; i > 0; i--)
    {
        for (j = 0; j < i; j++)
        {
            if (a[j] < a[j + 1])
            {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }
}

int main()
{
    int pid, i, n, a[50], j;
    printf("\nEnter the Number of elements: ");
    scanf("%d", &n);

    printf("\nEnter List of Numbers:\n");
    for (j = 0; j < n; j++)
    {
        scanf("%d", &a[j]);
    }

    pid = fork();

    if (pid == 0)
    {
        /* ----- CHILD PROCESS ----- */
        printf("\nI am Child Process, my pid is %d\n", getpid());
        printf("My Parent Process pid is %d\n", getppid());

        bubble_acs(a, n);
        printf("\nList of Numbers in Ascending Order:\n");
        for (j = 0; j < n; j++)
        {
            printf("%d ", a[j]);
        }
        printf("\n");

        /* Uncomment below for orphan demonstration */
        /*
        sleep(5);
        printf("\nAfter sleep, my new parent pid is %d (should be init)\n", getppid());
        */
    }
    else
    {
        /* ----- PARENT PROCESS ----- */
        /* Uncomment below to show zombie process temporarily */
        /*
        system("ps -el | grep Z");
        sleep(10);
        */

        i = wait(0); // wait for child to finish
        printf("\nPID of the terminated child process is %d\n", i);

        printf("\nI am Parent Process, my pid is %d\n", getpid());
        printf("I am Parent's Parent Process, my pid is %d\n", getppid());

        bubble_dcs(a, n);
        printf("\nList of Numbers in Descending Order:\n");
        for (j = 0; j < n; j++)
        {
            printf("%d ", a[j]);
        }
        printf("\n");
    }

    return 0;
}
