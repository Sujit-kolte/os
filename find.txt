db.students.find()

// 2. Find one document
db.students.findOne()

// 3. Find students with course "IT"
db.students.find({ course: "IT" })

// 4. Find students with age > 21
db.students.find({ age: { $gt: 21 } })

// 5. Find students with age between 20 and 22
db.students.find({ age: { $gte: 20, $lte: 22 } })

// 6. Find specific fields only
db.students.find({}, { name: 1, course: 1, _id: 0 })

// 7. Sort students by age ascending
db.students.find().sort({ age: 1 })

// 8. Limit results to 2
db.students.find().limit(2)

// 9. Count total students
db.students.countDocuments()

// 10. Find distinct course names
db.students.distinct("course")
