#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

#define SIZE 5   // size of the buffer

int buffer[SIZE];
int in = 0, out = 0;

sem_t empty, full;        // counting semaphores
pthread_mutex_t mutex;    // binary semaphore (mutex)

// Producer function
void *producer(void *arg) {
    int item = 0;
    while (1) {
        item++;  // produce next item
        sem_wait(&empty);               // wait if buffer is full
        pthread_mutex_lock(&mutex);     // lock critical section

        buffer[in] = item;
        printf("Produced item: %d\n", item);
        in = (in + 1) % SIZE;

        pthread_mutex_unlock(&mutex);   // unlock
        sem_post(&full);                // signal that item was produced

        sleep(1); // simulate time to produce
    }
}

// Consumer function
void *consumer(void *arg) {
    int item;
    while (1) {
        sem_wait(&full);                // wait if buffer is empty
        pthread_mutex_lock(&mutex);     // lock critical section

        item = buffer[out];
        printf("Consumed item: %d\n", item);
        out = (out + 1) % SIZE;

        pthread_mutex_unlock(&mutex);   // unlock
        sem_post(&empty);               // signal that one slot is free

        sleep(2); // simulate time to consume
    }
}

int main() {
    pthread_t prod, cons;

    // Initialize semaphores
    sem_init(&empty, 0, SIZE);
    sem_init(&full, 0, 0);
    pthread_mutex_init(&mutex, NULL);

    // Create producer and consumer threads
    pthread_create(&prod, NULL, producer, NULL);
    pthread_create(&cons, NULL, consumer, NULL);

    // Wait for both threads
    pthread_join(prod, NULL);
    pthread_join(cons, NULL);

    // Cleanup
    sem_destroy(&empty);
    sem_destroy(&full);
    pthread_mutex_destroy(&mutex);

    return 0;
}
